// Add duplicate meter number detection feature function checkDuplicateMeterNumber(meterNumber) { const journalData = JSON.parse(localStorage.getItem('journalDataPLC')) || {}; const today = new Date().toISOString().split('T')[0]; // Get today's date const entriesToday = journalData[today] || []; const duplicateEntry = entriesToday.find(entry => entry.meterNumber === meterNumber); if (duplicateEntry) { // Show modal warning with details alert(`❌ Номер закріплений\n\nAccount Number: ${duplicateEntry.accountNumber}\nTime: ${duplicateEntry.time}\nMeter Details: ${duplicateEntry.meterDetails}`); // Play beep sound const beep = new Audio('path/to/beep-sound.mp3'); // Add correct path to beep sound beep.play(); return true; // Duplicate found } return false; // No duplicate } // Sample usage in instNum input field const instNumInput = document.getElementById('instNum'); instNumInput.addEventListener('input', function() { const meterNumber = instNumInput.value; if (checkDuplicateMeterNumber(meterNumber)) { // Prevent further processing if duplicate instNumInput.value = ''; } });
